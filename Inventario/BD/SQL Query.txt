CREATE TABLE Proveedores(
	IdNit VARCHAR(15) UNIQUE NOT NULL,
	Razon_Social VARCHAR,
	Ciudad VARCHAR,
	PRIMARY KEY (IdNit)
);

CREATE TABLE "Productos" (
	"IdNitProv"	VARCHAR(15) NOT NULL,
	"Codigo"	VARCHAR(15) NOT NULL,
	"Descripcion"	VARCHAR,
	"Unidad"	VARCHAR(10),
	"Cantidad"	DOUBLE,
	"Precio"	DOUBLE,
	"Fecha"	DATE,
	FOREIGN KEY("IdNitProv") REFERENCES Proveedores(IdNit),
	PRIMARY KEY("IdNitProv","Codigo")
);
